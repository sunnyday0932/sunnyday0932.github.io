YoubikeRepository實作地方，使用JsonConvert.DeserializeObject反序列化拿回資料。

        /// <summary>
        /// 取得Youbike資料
        /// </summary>
        /// <returns></returns>
        /// <exception cref="HttpRequestException">取得Youbike 資訊失敗!</exception>
        public async Task<IEnumerable<YoubikeDataModel>> GetSationAsync()
        {

            var response = await this._httpClient.GetAsync(this._youbikeUrl);

            if (response.IsSuccessStatusCode)
            {
                var content = await response.Content.ReadAsStringAsync();
                var result = JsonConvert.DeserializeObject<IEnumerable<YoubikeDataModel>>(content);
                return result;
            }

            throw new HttpRequestException("取得Youbike 資訊失敗!");
        }

新增資料，使用Dapper一次大量INSERT。

        /// <summary>
        /// 新增Station
        /// </summary>
        /// <param name="conditions"></param>
        /// <returns></returns>
        public async Task<bool> InsertStationAsync(IEnumerable<StationCondition> conditions)
        {
            var sql = @"INSERT INTO YoubikeStation
                        (
	                        StationNo,
	                        StationName,
	                        Total,
	                        BikeAmount,
	                        StationArea,
	                        ModifyDate,
	                        Latitude,
	                        Longitude,
	                        Address,
	                        StationAreaEnglish,
	                        StationNameEnglish,
	                        AddressEnglish,
	                        Available,
	                        Active,
	                        SrcUpdateTime,
	                        UpdateTime,
	                        InfoTime,
	                        InfoDate
                        )
                        values
                        (
	                        @StationNo,
	                        @StationName,
	                        @Total,
	                        @BikeAmount,
	                        @StationArea,
	                        @ModifyDate,
	                        @Latitude,
	                        @Longitude,
	                        @Address,
	                        @StationAreaEnglish,
	                        @StationNameEnglish,
	                        @AddressEnglish,
	                        @Available,
	                        @Active,
	                        @SrcUpdateTime,
	                        @UpdateTime,
	                        @InfoTime,
	                        @InfoDate
                        )";
            var parameterList = new List<DynamicParameters>();
            conditions.ToList().ForEach(rows =>
                {
                    var parameters = new DynamicParameters();
                    parameters.Add("@StationNo", rows.StationNo, DbType.String);
                    parameters.Add("@StationName", rows.StationName, DbType.String);
                    parameters.Add("@Total", rows.Total, DbType.Int32);
                    parameters.Add("@BikeAmount", rows.BikeAmount, DbType.Int32);
                    parameters.Add("@StationArea", rows.StaionArea, DbType.String);
                    parameters.Add("@ModifyDate", rows.ModifyDate, DbType.DateTime);
                    parameters.Add("@Latitude", rows.Latitude, DbType.Double);
                    parameters.Add("@Longitude", rows.Longitude, DbType.Double);
                    parameters.Add("@Address", rows.Address, DbType.String);
                    parameters.Add("@StationAreaEnglish", rows.StationAreaEnglish, DbType.String);
                    parameters.Add("@StationNameEnglish", rows.StationNameEnglish, DbType.String);
                    parameters.Add("@AddressEnglish", rows.AddressEnglish, DbType.String);
                    parameters.Add("@Available", rows.Available, DbType.Int32);
                    parameters.Add("@Active", rows.Active, DbType.Int32);
                    parameters.Add("@SrcUpdateTime", rows.SrcUpdateTime, DbType.DateTime);
                    parameters.Add("@UpdateTime", rows.UpdateTime, DbType.DateTime);
                    parameters.Add("@InfoTime", rows.InfoTime, DbType.DateTime);
                    parameters.Add("@InfoDate", rows.InfoDate, DbType.DateTime);

                    parameterList.Add(parameters);
                });

            using (var conn = this._databaseHelper.GetConnection(this._connectionString))
            {
                var result = await conn.ExecuteAsync(
                    sql,
                    parameterList
                    );

                return result > 0;
            }
        }

StationService的地方有使用TransactionScope，主要用來避免執行一半錯誤發生資料不一致問題。

        /// <summary>
        /// 更新YouBike站
        /// </summary>
        /// <param name="stationInfo">The station information.</param>
        /// <returns></returns>
        public async Task<ResultDto> UpdateStationAsync(IEnumerable<StationInfoModel> stationInfo)
        {
            if (stationInfo is null ||
                stationInfo.Any().Equals(false))
            {
                return new ResultDto
                {
                    Sucess = false,
                    Message = "傳入資料為空"
                };
            }

            var condition = this._mapper.Map<IEnumerable<StationCondition>>(stationInfo);
            using (var scope = new TransactionScope(TransactionScopeAsyncFlowOption.Enabled))
            {
                await this._stationRepository.DeleteStationAsync();

                var insertResult = await this._stationRepository.InsertStationAsync(condition);

                if (insertResult.Equals(false))
                {
                    return new ResultDto
                    {
                        Sucess = false,
                        Message = "資料新增失敗"
                    };
                }
                scope.Complete();
                return new ResultDto
                {
                    Sucess = insertResult,
                    Message = "資料更新成功"
                };

            }

        }

Hangfire相關設定。
Youbike job會印出目前執行狀態以及步驟。

        /// <summary>
        /// 更新Youbike站資訊
        /// </summary>
        /// <param name="context">The context.</param>
        public async Task UpdateStation(PerformContext context)
        {
            var startMessage = $"排程起始時間：{DateTime.Now.ToLocalTime():yyyy/MM/dd HH:mm:ss}";
            context.WriteLine(startMessage);
            var stationInfo = await this._youbikeService.GetSationAsync();
            context.WriteLine($"抓取YouBike 2.0資料共 {stationInfo.Count()} 筆");

            context.WriteLine($"開始更新資料庫");
            var stationInfoModel = this._mapper.Map<IEnumerable<StationInfoModel>>(stationInfo);
            await this._stationService.UpdateStationAsync(stationInfoModel);
            context.WriteLine($"更新資料庫完成");

            var endMessage = $"排程結束時間：{DateTime.Now.ToLocalTime():yyyy/MM/dd HH:mm:ss}";
            context.WriteLine(endMessage);
        }

Hangfire Trigger，一樣起始先把Job清空，配合原始API更新時間，一分鐘抓取資料更新即可。

        public void OnStart()
        {
            //清除所有的排程
            using var conn = JobStorage.Current.GetConnection();
            conn.GetRecurringJobs().ForEach(row =>
            {
                RecurringJob.RemoveIfExists(row.Id);
            });

            //YouBike 2.0 排程 每分鐘執行一次
            RecurringJob.AddOrUpdate<IYouBikeJob>
                (
                    methodCall: x => x.UpdateStation(null),
                    cronExpression: "*/1 * * * *",
                    timeZone: TimeZoneInfo.Local
                );
        }

StartUp，特別的地方是為了避免sockets exhaustion問題將HttpClient註冊為Singleton共用。

        /// <summary>
        /// 取得DB連線
        /// </summary>
        /// <returns></returns>
        private string GetConnection(string databaseName)
        {
            var connection = this.Configuration.GetValue<string>($"ConnectionString:{databaseName}");

            return connection;
        }

        /// <summary>
        /// 取得Url
        /// </summary>
        /// <returns></returns>
        private string GetUrl(string urlName)
        {
            var url = this.Configuration.GetValue<string>($"Url:{urlName}");

            return url;
        }
        
         public void ConfigureServices(IServiceCollection services)
        {
            services.AddControllersWithViews();

            //DI Extensions
            services.AddDependencyInjection();

            //HttpClient
            services.AddSingleton<HttpClient>(new HttpClient());

            //Url
            var youbikeUrl = this.GetUrl("Youbike");

            //Connection
            var hangfireConnection = this.Configuration.GetConnectionString(name: "Hangfire");
            var northWindConnection = this.Configuration.GetConnectionString(name: "Northwind");

            //Repository
            services.AddSingleton<IDatabaseHelper, DatabaseHelper>();
            services.AddScoped<IYoubikeRepository>(sp =>
            {
                return new YoubikeRepository(
                    youbikeUrl,
                    sp.GetRequiredService<HttpClient>());
            });

            services.AddScoped<IStationRepository>(sp =>
            {
                return new StationRepository(
                    northWindConnection,
                    sp.GetRequiredService<IDatabaseHelper>());
            });

            //AutoMapper
            services.AddAutoMapper(
                typeof(ControllerProfile),
                typeof(ServiceProfile));

            //Hangfire
            services.AddHangfire(config =>
            {
                config.UseSqlServerStorage(
                    nameOrConnectionString: hangfireConnection,
                    options: new SqlServerStorageOptions
                    {
                        SchemaName = "YouBikeScheduler",
                        //Job資料的保存期限
                        JobExpirationCheckInterval = TimeSpan.FromMinutes(value: 60)
                    })
                //顯示log
                .UseConsole()
                //使用中的連線
                .UseDashboardMetric(SqlServerStorage.ActiveConnections)
                //連線數量
                .UseDashboardMetric(SqlServerStorage.TotalConnections)
                //Hangfire Server數量
                .UseDashboardMetric(DashboardMetrics.ServerCount)
                //Recurring Job數量
                .UseDashboardMetric(DashboardMetrics.RecurringJobCount)
                //排程數量
                .UseDashboardMetric(DashboardMetrics.ScheduledCount)
                //目前在Queue中的排程數量
                .UseDashboardMetric(DashboardMetrics.EnqueuedAndQueueCount)
                //處理中的排程數量
                .UseDashboardMetric(DashboardMetrics.ProcessingCount)
                //正在等待處理中的排程數量
                .UseDashboardMetric(DashboardMetrics.AwaitingCount)
                //已被刪除的排程數量
                .UseDashboardMetric(DashboardMetrics.DeletedCount)
                //執行失敗的排程數量
                .UseDashboardMetric(DashboardMetrics.FailedCount)
                //成功執行的排程數量
                .UseDashboardMetric(DashboardMetrics.SucceededCount);
            });

            services.AddHangfireServer(options =>
            {
                //Server Name
                options.ServerName = $"{Environment.MachineName}:NetCore";
            });
        }
